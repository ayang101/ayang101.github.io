    class Solution {
    public:
    int romanToInt(string s) {
    int sum = 0;
    if(s.length() == 0){
    return sum;
    }
    for(int i=0; i<s.length(); i++){
    if(s[i] == 'I'){
    if(s[i+1] == 'V' && i<s.length()){
    sum += 4;
    i++;
    }else if(s[i+1] == 'X' && i<s.length()){
    sum += 9;
    i++;
    }else{
    sum += 1;
    }
    }else if(s[i] == 'X'){
    if(s[i+1] == 'L' && i<s.length()){
    sum += 40;
    i++;
    }else if(s[i+1] == 'C' && i<s.length()){
    sum += 90;
    i++;
    }else{
    sum += 10;
    }
    }else if(s[i] == 'C'){
    if(s[i+1] == 'D' && i<s.length()){
    sum += 400;
    i++;
    }else if(s[i+1] == 'M' && i<s.length()){
    sum += 900;
    i++;
    }else{
    sum += 100;
    }
    }else if(s[i] == 'V'){
    sum += 5;
    }else if(s[i] == 'L'){
    sum += 50;
    }else if(s[i] == 'D'){
    sum += 500;
    }else if(s[i] == 'M'){
    sum += 1000;
    }
    }return sum;
    }
    };
